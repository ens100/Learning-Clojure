 (defn add
  ([ingredient]

   (cond
     (squeezed? ingredient)

     (scooped? ingredient)
     (add-scooped ingredient amount)
     (simple? ingredient)
     (add-simple ingredient amount)
     :else
     (do
       (println "I do not know the ingredient" ingredient)
       :error))))
